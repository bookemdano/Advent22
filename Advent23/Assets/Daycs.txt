using AoCLibrary;
namespace Advent23
{
	internal class Day : IDayRunner
	{
		public bool IsReal => false;

		private object? Star1()
		{
			var rv = 0L;
			var lines = Program.GetLines(StarEnum.Star1, IsReal);
            if (!IsReal)
                Utils.Assert(rv, 0L);
			return rv;
		}
		private object? Star2()
		{
			var rv = 0L;
			var lines = Program.GetLines(StarEnum.Star2, IsReal);
            if (!IsReal)
                Utils.Assert(rv, 0L);
			return rv;
		}
		public RunnerResult Run()
		{
			var rv = new RunnerResult();
			rv.Star1 = Star1();
			rv.Star2 = Star2();
			return rv;
		}
	}
}
